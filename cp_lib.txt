	function cp_get($url, $refer="", $auto=0){
		return cp_post($url, array(), $refer, 0, $auto);
	}

	function cp_post($url, $fields, $refer="", $post=1, $auto=0){
		global $g_cookies, $g_lasturl;

		if ($auto){$refer = $g_lasturl;}
		$g_lasturl = $url;

		#$g_cookies['fcP'] = 'C=6&T='.(time() * 1000);

		$curl = curl_init();

		curl_setopt($curl, CURLOPT_HEADER, 1);
		curl_setopt($curl, CURLOPT_VERBOSE, 1);
		curl_setopt($curl, CURLOPT_RETURNTRANSFER, 1);
		curl_setopt($curl, CURLOPT_URL, $url);
		curl_setopt($curl, CURLOPT_POST, $post);

		if ($post){
			curl_setopt($curl, CURLOPT_POSTFIELDS, $fields);
		}

		if (count($g_cookies) > 0){
			$cookies = array();
			foreach(array_keys($g_cookies) as $k){
				$cookies[] = urlencode($k)."=".urlencode($g_cookies[$k]);
			}
			$cookies = implode("; ", $cookies);
			curl_setopt($curl, CURLOPT_COOKIE, $cookies);
		}

		if ($refer){
			curl_setopt($curl, CURLOPT_REFERER, $refer);
		}

		curl_setopt($curl, CURLOPT_USERAGENT, "Mozilla/4.0 (compatible; MSIE 6.0; Windows NT 4.0)");

		$data = curl_exec($curl);
		if ($error = curl_error($curl)){
			echo "CURL ERROR: $error<br>";
		}

		curl_close($curl);

		$data = str_replace("\r", "", $data);

		$blocks = explode("\n\n", $data);
		$head = array_shift($blocks);

		$ok = 1;
		do {
			$block = array_shift($blocks);
			if (preg_match("/^HTTP\//",$block)){
				$head .= "\n\n".$block;
			}else{
				array_unshift($blocks, $block);
				$ok = 0;
			}
		} while($ok);

		$body = implode("\n\n", $blocks);

		$method = $post?'POST':'GET';

		#echo "<b>Got page via $method:</b> <i>$url</i><br>";

		return array($head, $body);
	}

	function cp_parse_cookies($head){
		global $g_cookies;

		if (preg_match_all("/^Set-Cookie: (.*)$/m", $head, $matches, PREG_SET_ORDER)){
			foreach($matches as $match){
				$chunks = explode(";", $match[1]);
				$chunk = trim(array_shift($chunks));
				list($k,$v) = explode("=", $chunk, 2);
				$g_cookies[$k] = $v;
				#echo "FOUND COOKIE $k ($v)<br><br>";
			}
		}

	}
